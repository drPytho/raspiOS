15   /** 
16    * @description 
17    *    gathers all the fils in a number of folders 
18    * @param [0] paths 
19    *    An array of paths to search through 
20    * @param [1] recursive 
21    *    If it should be a recursive search or not 
22    * @return 
23    *    A vector with all the files in it  
24    * @throw 
25    *    Nothing 
26    * 
27    * @note 
28    *    Nees to check for possible memmory leaks with reapeted use. 
29    */

31       std::vector<std::string> getFilesInFolder(std::string paths[], bool recursive) 
32       { 
33           std::vector<std::string> Files; 
34           if(sizeof(paths) == 0)                                                      //Make sure we were provided some paths 
35               return Files; 
36    
37           std::vector<std::string> v_paths(paths, paths + sizeof(paths)/sizeof(paths[0])); 
38    
39           int searchPaths = v_paths.size();                                           //Get the number of paths 
40            
41           DIR *dp; 
42           struct dirent *dirp; 
43            
44    
45           for(int i = 0; i < searchPaths; i++)                                        //Loop through the paths 
46           { 
47               if((dp = opendir(paths[i].c_str())) == NULL)                                    //Directory do not exist.... Or something else 
48                   continue;                                                           //Move on to the next path 
49    
50               while((dirp = readdir(dp)) != NULL)                                     //Aslong as we get a new dirent structure there is a new file there with a name on it 
51               { 
52                   if(dirp->d_type == 0x4)                                             //if it's a folder 
53                   { 
54                       if(recursive)                                                   //if it's a recursive search 
55                       { 
56                           v_paths.push_back(v_paths[i] + "/" + std::string(dirp->d_name));    //add the dir to the directories to search 
57                           searchPaths++; 
58                       } 
59                   }else{                                                              //Othervise 
60                       Files.push_back(v_paths[i] + "/" + std::string(dirp->d_name));      //Get the god damn name 
61                   } 
62               } 
63           } 
64    
65           return Files;                                                               //Pass it forward :) 
66       }

68       fType getFileType(char* extention) 
69       { 
70           //Lowaecase extention 
71           for(int i = 0; i < (VIDEO_SIZE + AUDIO_SIZE + EMULATOR_SIZE + GRAPHICS_SIZE); i++) 
72           { 
73               if (i < VIDEO_SIZE) 
74               { 
75                   if(!strcmp(VIDEO[i], extention)) 
76                       return LPI::D_VIDEO; 
77               }else if(i < (AUDIO_SIZE + VIDEO_SIZE)) 
78               { 
79                   if(!strcmp(AUDIO[i - VIDEO_SIZE], extention)) 
80                       return LPI::D_AUDIO; 
81               }else if(i < (AUDIO_SIZE + VIDEO_SIZE + EMULATOR_SIZE)) 
82               { 
83                   if(!strcmp(EMULATOR[i - (VIDEO_SIZE + AUDIO_SIZE)], extention)) 
84                       return LPI::D_EMULATOR; 
85               }else if(i < (AUDIO_SIZE + VIDEO_SIZE + EMULATOR_SIZE)) 
86               { 
87                   if(!strcmp(GRAPHICS[i - (VIDEO_SIZE + AUDIO_SIZE + EMULATOR_SIZE)], extention)) 
88                       return LPI::D_GRAPHICS; 
89               } 
90           } 
91           return LPI::D_OTHER; 
92       }
